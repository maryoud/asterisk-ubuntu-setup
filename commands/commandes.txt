# Voici les commnades nécessaires pour l'installation et la configuration d'Asterisk

#installer asterisk:

 sudo install asterisk -y

#vérification de l'état:

 sudo systemctl status asterisk

#activation :

 sudo systemctl start asterisk 

#création de backup:

 sudo mv sip.conf sip.conf.backup     #( même pour les autres)


#le serveur:

 sudo asterisk -r

#redémarrer asterisk:

  reload

#Voir les lignes:

 sip show peers
