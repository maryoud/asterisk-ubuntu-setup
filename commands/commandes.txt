#configuration des fichiers:    3 fichiers (sip.conf, extensions.conf et voicemail.conf)

#installer asterisk:
 sudo install asterisk -y
#vérification de l'état:
 sudo systemctl status asterisk
#activation :
 sudo systemctl start asterisk 
#création de backup:
 sudo mv sip.conf sip.conf.backup  
#( même pour les autres)
#
configuration: sudo gedit sip.conf
# (emplacement  /etc/asterisk/ )  (même pour les autres)
#vérification:
 sudo asterisk -r
#redémarrer asterisk:
  reload
#verification:
 sip show peers
